Design Tokens
Variáveis CSS globais definidas em :root. Nenhum valor hardcoded no código — sempre referenciar os tokens.
/* globals.css ou _tokens.css — importar no _app.tsx */ :root { /* Brand */ --gold: #BFA06A; --gold-light: #D4B87E; --gold-dim: #8B7145; --gold-bg: rgba(191,160,106,0.08); --gold-border: rgba(191,160,106,0.20); /* Canvas */ --bg-void: #080809; /* body background */ --bg-base: #0F0F11; /* app shell / inputs */ --bg-surface: #131316; /* cards */ --bg-elevated: #1A1A1E; /* dialogs / dropdowns */ --bg-nav: #0A0A0C; /* topbar / leftnav */ --bg-hover: rgba(255,255,255,0.03); --bg-active: rgba(191,160,106,0.08); /* Borders */ --border-sub: rgba(255,255,255,0.04); --border-low: rgba(255,255,255,0.06); --border-mid: rgba(255,255,255,0.10); --border-high: rgba(255,255,255,0.16); /* Text */ --text-primary: #F0EBE3; --text-secondary:#A09A94; --text-muted: #4a4a54; --text-disabled: #2a2a32; /* Semantic */ --green: #4CAF82; --red: #E05252; --amber: #F0A040; --blue: #4A9EFF; --purple: #9C6FDE; /* Typography */ --font-display: 'Cormorant Garamond', serif; --font-body: 'Inter', sans-serif; --font-mono: 'JetBrains Mono', monospace; /* Radii */ --r-sm: 6px; --r-md: 8px; --r-lg: 12px; --r-xl: 16px; /* Transitions */ --t-fast: 120ms ease; --t-base: 180ms ease; --t-slow: 260ms ease; }

Animações
/* ───────────────────────────────────────────── TOKENS DE TRANSIÇÃO ───────────────────────────────────────────── */ --t-fast: 120ms ease // hover states, border changes --t-base: 180ms ease // button press, toggle, dialog open --t-slow: 260ms ease // slide-over, card hover, chart update /* ───────────────────────────────────────────── KEYFRAMES GLOBAIS ───────────────────────────────────────────── */ @keyframes fadeIn { from { opacity: 0 } to { opacity: 1 } } @keyframes slideUp { from { transform: translateY(8px); opacity: 0 } to { transform: translateY(0); opacity: 1 } } @keyframes scaleIn { from { transform: scale(0.96); opacity: 0 } to { transform: scale(1); opacity: 1 } } @keyframes shimmer { from { background-position: -400px 0 } to { background-position: 400px 0 } } /* ───────────────────────────────────────────── USO POR CONTEXTO ───────────────────────────────────────────── */ Dialog open → scaleIn 180ms ease Sheet open → translateX(100%)→0, 260ms ease Dropdown open → slideUp 120ms ease Toast enter → slideUp 200ms ease Page transition → fadeIn 160ms ease Kanban card DnD → box-shadow 0 8px 24px rgba(0,0,0,0.4) enquanto dragging Progress bar → width transition 400ms ease (na primeira renderização) Chart bars → height/opacity 300ms ease (na montagem do componente) /* ───────────────────────────────────────────── REGRAS ───────────────────────────────────────────── */ // NÃO animar layout shifts (height, margin, padding) // SEMPRE usar transform e opacity para performance (GPU) // NÃO usar animation em elementos que repetem muito (table rows) // Respeitar prefers-reduced-motion: @media (prefers-reduced-motion: reduce) { *, ::before, ::after { animation-duration: 0.01ms !important; transition-duration: 0.01ms !important; } }
